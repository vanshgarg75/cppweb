13. Program to right rotate the elements of an array

source code

          #include < iostream >
using namespace std;

// Function to right rotate the elements of an array by one position
void rightRotateByOne(int arr[], int size) {
    int temp = arr[size - 1];
    for (int i = size - 1; i > 0; i--) {
        arr[i] = arr[i - 1];
    }
    arr[0] = temp;
}

// Function to perform right rotation of an array by k positions
void rightRotate(int arr[], int size, int k) {
    for (int i = 0; i < k; i++) {
        rightRotateByOne(arr, size);
    }
}

// Function to print the elements of an array
void printArray(int arr[], int size) {
    for (int i = 0; i < size; i++) {
        cout << arr[i] << " ";
    }
    cout << endl;
}

int main() {
    int arr[] = {1, 2, 3, 4, 5};
    int size = sizeof(arr) / sizeof(arr[0]);
    int k = 2; // Number of positions to rotate

    cout << "Original Array: ";
    printArray(arr, size);

    // Right rotate the array by k positions
    rightRotate(arr, size, k);

    cout << "Array after right rotation by " << k << " positions: ";
    printArray(arr, size);

    return 0;
}

14. Program to sort the elements of an array in ascending order

using namespace std;

// Function to sort the elements of an array in ascending order
void bubbleSortAscending(int arr[], int size) {
    for (int i = 0; i < size - 1; i++) {
        for (int j = 0; j < size - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                // Swap elements if they are in the wrong order
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

// Function to print the elements of an array
void printArray(const int arr[], int size) {
    for (int i = 0; i < size; i++) {
        cout << arr[i] << " ";
    }
    cout << endl;
}

int main() {
    int arr[] = {5, 3, 9, 2, 7};
    int size = sizeof(arr) / sizeof(arr[0]);

    cout << "Original Array: ";
    printArray(arr, size);

    // Call the function to sort the array in ascending order
    bubbleSortAscending(arr, size);

    cout << "Array sorted in ascending order: ";
    printArray(arr, size);

    return 0;
}

16. Program to Find 3rd Largest Number in an array

source code

          #include < iostream >
using namespace std;

// Function to find the third largest number in an array
int findThirdLargest(const int arr[], int size) {
    // If the array has less than 3 elements, return a sentinel value
    if (size < 3) {
        return -1; // Return some sentinel value indicating error or absence of valid result
    }

    // Initialize three variables to keep track of the three largest numbers
    int firstLargest = arr[0];
    int secondLargest = arr[1];
    int thirdLargest = arr[2];

    // Arrange the largest variables in decreasing order
    if (secondLargest > firstLargest) {
        swap(firstLargest, secondLargest);
    }
    if (thirdLargest > secondLargest) {
        swap(secondLargest, thirdLargest);
    }
    if (secondLargest > firstLargest) {
        swap(firstLargest, secondLargest);
    }

    // Iterate through the array to find the three largest numbers
    for (int i = 3; i < size; i++) {
        if (arr[i] > firstLargest) {
            thirdLargest = secondLargest;
            secondLargest = firstLargest;
            firstLargest = arr[i];
        } else if (arr[i] > secondLargest && arr[i] != firstLargest) {
            thirdLargest = secondLargest;
            secondLargest = arr[i];
        } else if (arr[i] > thirdLargest && arr[i] != secondLargest && arr[i] != firstLargest) {
            thirdLargest = arr[i];
        }
    }

    return thirdLargest;
}

int main() {
    int arr[] = {5, 3, 9, 2, 7};
    int size = sizeof(arr) / sizeof(arr[0]);

    // Call the function to find the third largest number
    int thirdLargest = findThirdLargest(arr, size);

        cout << "The third largest number in the array is: " << thirdLargest << endl;

21. Program to Remove Duplicate Element in an array

source code

          #include < iostream >
using namespace std;

// Function to remove duplicate elements from an array
int removeDuplicates(int arr[], int size) {
    if (size == 0 || size == 1) {
        return size;
    }

    int index = 0;

    // Traverse the array and move non-duplicate elements to the beginning
    for (int i = 0; i < size - 1; i++) {
        if (arr[i] != arr[i + 1]) {
            arr[index++] = arr[i];
        }
    }

    // Store the last element (which is always unique)
    arr[index++] = arr[size - 1];

    return index; // Return the new size of the array
}

// Function to print the array
void printArray(int arr[], int size) {
    for (int i = 0; i < size; i++) {
        cout << arr[i] << " ";
    }
    cout << endl;
}

int main() {
    int arr[] = {1, 2, 2, 3, 4, 4, 5};
    int size = sizeof(arr) / sizeof(arr[0]);

    cout << "Original Array: ";
    printArray(arr, size);
    int newSize = removeDuplicates(arr, size);    //function call to remove duplicates

    cout << "Array after removing duplicates: ";
    printArray(arr, newSize);

    return 0;
}

23. Write a C++ program to swap the first and last elements of an array and create a new array

source code

          #include < iostream >
          using namespace std;

// Function to swap the first and last elements of an array and create a new array
int* swapFirstLast(const int arr[], int size) {
    int* newArr = new int[size]; // Allocate memory for new array
    
    // Copy the original array to the new array
    for (int i = 0; i < size; i++) {
        newArr[i] = arr[i];
    }

    // Swap the first and last elements
    int temp = newArr[0];
    newArr[0] = newArr[size - 1];
    newArr[size - 1] = temp;

    return newArr; // Return the new array
}

// function to print the array
void printArray(const int arr[], int size) {
    for (int i = 0; i < size; i++) {
        cout << arr[i] << " ";
    }
    cout << endl;
}

int main() {
    int arr[] = {1, 2, 3, 4, 5};
    int size = sizeof(arr) / sizeof(arr[0]);

    cout << "Original Array: ";
    printArray(arr, size);

    // Call the function to swap first and last elements and create a new array
    int* swappedArr = swapFirstLast(arr, size);

    cout << "Array after swapping first and last elements: ";
    printArray(swappedArr, size);

    // Don't forget to deallocate memory for the new array
    delete[] swappedArr;

    return 0;
}
