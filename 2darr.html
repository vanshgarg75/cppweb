3. Program to Multiply Two Matrices

source code

#include<iostream>
            using namespace std;
            
            int main()
            {
              int a[10][10],b[10][10],c[10][10],i,j,k,r1,c1,r2,c2;
              cout<<"\nEnter number of rows and columns of 1st matrix";
              cin>>r1>>c1;
              cout<<"\nEnter number of rows and columns of 2nd matrix";
              cin>>r2>>c2;
              if(c1==r2)
              {
                cout<<"\nEnter elements of 1st matrix";
                for(i=0;i<r1;i++)
                  for(j=0;j<c1;j++)
                    cin>>a[i][j];
                cout<<"\nEnter elements of 2nd matrix";
                for(i=0;i<r2;i++)
                  for(j=0;j<c2;j++)
                    cin>>b[i][j];
              //calculate product of two matrix//
              for(i=0;i<r1;i++)
              {
                for(j=0;j<c2;j++)
                {
                c[i][j]=0;
                  for(k=0;k<c1;k++)
                    c[i][j]=c[i][j]+a[i][k]*b[k][j];
                }
              }
              cout<<"Product of two matrix: "<<endl;
              for(i=0;i<r1;i++)
              {
                for(j=0;j<c2;j++)
                  cout<<"\t"<<c[i][j];
                cout<<"\n";
              }
            }
            else
              cout<<"\n Means Multipliction is not possible";
              return 0;
            }

8.frequency of even and odd elements of matrix

        

        

          
source code

          
            #include <iostream>
              using namespace std;
              
              int main() 
              {
                
                  int matrix[3][3];
                  int oddCount = 0, evenCount = 0;
              
                  cout << "Enter 9 elements of the 3x3 matrix:" << endl;
              
                  // Input elements of the matrix
                  for (int i = 0; i < 3; ++i) {
                      for (int j = 0; j < 3; ++j) {
                          cin >> matrix[i][j];
                      }
                  }
              
                  // Counting even and odd numbers
                  for (int i = 0; i < 3; ++i) {
                      for (int j = 0; j < 3; ++j) {
                          if (matrix[i][j] % 2 == 0)
                              evenCount++;
                          else
                              oddCount++;
                      }
                  }
              
                  // Displaying the results
                  cout << "Frequency of even numbers: " << evenCount << endl;
                  cout << "Frequency of odd numbers: " << oddCount << endl;
              
                  return 0;
              }
