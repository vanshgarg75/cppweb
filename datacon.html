11.C++ Program to convert Binary to Decimal

source code

            #include< iostream>
           using namespace std;
            //function to convert binary to decimal
            int binaryToDecimal(int n)
            {
                int num=n;
                int dec_value=0;
            
                //intializing base value to 1,i.e 2^0
                int base=1;
                int temp=num;
                while(temp)
                {
                    int last_digit=temp%10;
                    temp=temp/10;
            
                    dec_value+=last_digit*base;
                    base=base*2;
                }
                return dec_value;
            }
            //main definition'
            int main()
            {
                int num;
               
                cout<<"Enter binary no : ";
                cin>>num;
                cout<<"The equivalent decimal no to "<< num<<"is : ";
                cout<< binaryToDecimal(num)<< endl;
                
                return 0;
            }

output

Enter binary no : 1010
            The equivalent decimal no to 1010is : 10

12. C++ Program to convert Decimal to Binary

source code

            
#include< iostream>
using namespace std;
//function to convert a decimal to binary number
void decToBinary(int n)
{
	//array to store binary number
	int binaryNum[32];

	//counter for binary array
	int i=0;
	while(n>0)
	{
		//storing remainder in binary array
		binaryNum[i]=n%2;
		n=n/2;
		i++;
	}

	//printint binary array in reverse order
	for(int j=i-1;j>=0;j--)
		cout<< binaryNum[j];
}
int main()
{
	int n;
	
	cout<<"Enter the decimal no : ";
	cin>>n;
	cout<<"The equivalent binary no to "<< n<<" is : ";
	decToBinary(n);
	
	return 0;
}

output

Enter the decimal no : 10
            The equivalent binary no to 10 is : 1010

13. C++ Program to convert Hex to Decimal

source code

            
            #include< iostream>
            #include< cstring>
            #include< cmath>
            using namespace std;
            
            int main()
            {
                char num[20];
                int i,r,len,hex=0;
               
                cout<<"Enter a hexadecimal number : ";
                cin>>num;
                len=strlen(num);
                for(i=0;num[i]!='\0';i++)
                {
                    len--;
                    if(num[i]>='0'&&num[i]<='9')
                        r=num[i]-48;
                    else if(num[i]>='a'&&num[i]<='f')
                        r=num[i]-87;
                    else if(num[i]>='A'&&num[i]<='F')
                        r=num[i]-55;
                    hex+=r*pow(16,len);
                }
                cout<<"\nDecimal equivalent of "<< num<<" is : "<< hex;
                
                return 0;
            }

output

Enter a hexadecimal number : a

            Decimal equivalent of a is : 10

15. C++ Program to convert Octal to Decimal

source code

            #include<iostream>
            using namespace std;
            //function to convert octal to decimal
            int octalToDecimal(int n)
            {
              int num=n;
              int dec_value=0;
              //initializing base value to 1,i.e 8^0
              int base=1;
              int temp=num;
              while(temp)
              {
                //extracting last digit
                int last_digit=temp%10;
                temp=temp/10;
                /*Multiplying last digit with appropriate base value and
                   adding it to dec_value*/
                dec_value+=last_digit*base;
                base=base*8;
              }
                return dec_value;
            }
            //main definition
            int main()
            {	int num;	
              cout<<"Enter the octal no: ";
              cin>>num;
              cout<<"the decimal value of no is :"<< octalToDecimal(num)<< endl;	
              return 0;
            }

output

            Enter the octal no: 7
the decimal value of no is :7

            void decimalToOctal(int deciNum)
            {
              int octalNum=0,countval=1;
              while(deciNum!=0)
              {
                //decimals remainder is calculated
                int remainder=deciNum%8;
                //storing the octal value
                octalNum+=remainder*countval;
                //storing exponential value
