#include < iostream >
#include < string.h >
using namespace std;

// Class representing an account
class Account {
private:
    char accountNumber[20];
    char accountHolderName[50];
    double balance;

public:
    // Constructor
    Account(const char* accNumber, const char* accHolderName, double initialBalance) {
        strcpy(accountNumber, accNumber);
        strcpy(accountHolderName, accHolderName);
        balance = initialBalance;
    }

    // Method to deposit money
    void deposit(double amount) {
        balance += amount;
        cout << "Deposit of $" << amount << " successful. New balance: $" << balance << endl;
    }

    // Method to withdraw money
    void withdraw(double amount) {
        if (balance >= amount) {
            balance -= amount;
            cout << "Withdrawal of $" << amount << " successful. New balance: $" << balance << endl;
        } else {
            cout << "Insufficient funds. Withdrawal failed." << endl;
        }
    }

    // Getter method for account number
    const char* getAccountNumber() {
        return accountNumber;
    }

    // Getter method for account holder name
    const char* getAccountHolderName() {
        return accountHolderName;
    }

    // Getter method for balance
    double getBalance() {
        return balance;
    }
};

// Class representing a bank
class Bank {
private:
    static const int MAX_ACCOUNTS = 100; // Maximum number of accounts
    Account* accounts[MAX_ACCOUNTS]; // Array to store pointers to accounts
    int numAccounts; // Number of accounts in the bank

public:
    // Constructor
    Bank() : numAccounts(0) {}

    // Method to add an account
    void addAccount(Account* account) {
        if (numAccounts < MAX_ACCOUNTS) {
            accounts[numAccounts++] = account;
            cout << "Account added successfully." << endl;
        } else {
            cout << "Maximum number of accounts reached. Cannot add more accounts." << endl;
        }
    }

    // Method to remove an account
    void removeAccount(const char* accNumber) {
        for (int i = 0; i < numAccounts; ++i) {
            if (strcmp(accounts[i]->getAccountNumber(), accNumber) == 0) {
                delete accounts[i]; // Free memory
                // Move accounts after the removed one to fill the gap
                for (int j = i; j < numAccounts - 1; ++j) {
                    accounts[j] = accounts[j + 1];
                }
                numAccounts--;
                cout << "Account removed successfully." << endl;
                return;
            }
        }
        cout << "Account not found. Removal failed." << endl;
    }

    // Method to deposit money into an account
    void depositToAccount(const char* accNumber, double amount) {
        for (int i = 0; i < numAccounts; ++i) {
            if (strcmp(accounts[i]->getAccountNumber(), accNumber) == 0) {
                accounts[i]->deposit(amount);
                return;
            }
        }
        cout << "Account not found. Deposit failed." << endl;
    }

    // Method to withdraw money from an account
    void withdrawFromAccount(const char* accNumber, double amount) {
        for (int i = 0; i < numAccounts; ++i) {
            if (strcmp(accounts[i]->getAccountNumber(), accNumber) == 0) {
                accounts[i]->withdraw(amount);
                return;
            }
        }
        cout << "Account not found. Withdrawal failed." << endl;
    }
};

int main() {
    // Creating a bank object
    Bank bank;

    // Adding some accounts to the bank
    bank.addAccount(new Account("123456", "nav", 1000));
    bank.addAccount(new Account("789012", "niharika", 2000));

    // Depositing and withdrawing money from accounts
    bank.depositToAccount("123456", 500);
    bank.withdrawFromAccount("789012", 1000);

    // Attempting to remove an account
    bank.removeAccount("123456");

    // Attempting to deposit to a non-existent account
    bank.depositToAccount("111111", 500);
