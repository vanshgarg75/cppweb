1. Program to Check Anagram Explanation Two Strings are called the anagram if they contain the same characters. However, the order or sequence of the characters can be different.

source code

            #include < iostream >
#include < string.h >

using namespace std;

 Function to check if two strings are anagrams
int areAnagrams(const char* str1, const char* str2) {
     If the lengths of the strings are different, they cannot be anagrams
    if (strlen(str1) != strlen(str2))
        return 0;

     Initialize an array to store character counts
    const int CHAR_COUNT = 256;  Assuming ASCII characters
    int charCount[CHAR_COUNT] = {0};

    for (int i = 0; str1[i] != '\0'; i++)
        charCount[str1[i]]++;

    for (int i = 0; str2[i] != '\0'; i++)
        charCount[str2[i]]--;

     Check if all counts are zero
    for (int i = 0; i < CHAR_COUNT; i++) {
        if (charCount[i] != 0)
            return 0;
    }

    return 1;
}

int main() {
    char str1[100], str2[100];

     Input two strings from the user
    cout << "Enter the first string: ";
    cin >> str1;
    cout << "Enter the second string: ";
    cin >> str2;

     Check if the strings are anagrams
    if (areAnagrams(str1, str2))
        cout << "The strings are anagrams." << endl;
    else
        cout << "The strings are not anagrams." << endl;

    return 0;
}

4. Program to perform Deep Copy for String

source code

            #include < iostream>
#include < cstring>

using namespace std;

class MyString {
private:
    char *str;  Pointer to character array to hold the string
    int length;  Length of the string

public:
     Constructor
    MyString(const char *s = "") {
        length = strlen(s);
        str = new char[length + 1];
        strcpy(str, s);
    }

     Destructor
    ~MyString() {
        delete[] str;
    }

     Deep copy constructor
    MyString(const MyString &other) {
        length = other.length;
        str = new char[length + 1];
        strcpy(str, other.str);
    }

     Method to display the string
    void display() {
        cout << "String: " << str << endl;
    }
};

int main() {
    MyString str1("Hello");
    MyString str2 = str1;  Deep copy performed

    cout << "Original String: ";
    str1.display();
    cout << "Copied String: ";
    str2.display();

    return 0;
}

7.Program to check whether a string is a Palindrome.

source code

            #include < iostream>
                #include < string.h>
                using namespace std;
                
                 Function to check if a string is a palindrome
                int isPalindrome(char str[]) {
                    int length = strlen(str);
                    int i, j;
                
                     Remove spaces and punctuation marks
                    for (i = 0, j = 0; i < length; i++) {
                        if (isalnum(str[i])) {
                            str[j++] = tolower(str[i]);
                        }
                    }
                    str[j] = '\0';  Null terminate the modified string
                
                     Check for palindrome
                    for (i = 0, j = strlen(str) - 1; i < j; i++, j--) {
                        if (str[i] != str[j]) {
                            return 0;  Not a palindrome
                        }
                    }
                    return 1;  Palindrome
                }

24. Program to find the longest repeating sequence in a string

source code

#include <iostream>
            #include <string>
            #include <vector>
          
            std::string longest_repeating_sequence(const std::string& str) {
                int n = str.length();
                std::vector< std::vector< int>> dp(n + 1, std::vector< int>(n + 1, 0));
            
                int max_length = 0;
                int end_index = 0;
            
                for (int i = 1; i <= n; ++i) {
                    for (int j = i + 1; j <= n; ++j) {
                        if (str[i - 1] == str[j - 1] && dp[i - 1][j - 1] < (j - i)) {
                            dp[i][j] = dp[i - 1][j - 1] + 1;
            
                            if (dp[i][j] > max_length) {
                                max_length = dp[i][j];
                                end_index = i - 1;
                            }
                        } else {
                            dp[i][j] = 0;
                        }
                    }
                }
            
                if (max_length > 0) {
                    return str.substr(end_index - max_length + 1, max_length);
                } else {
                    return "No repeating sequence found.";
                }
            }
            
            int main() {
                std::string input_string;
            
                // Input string
                std::cout << "Enter a string: ";
                std::cin >> input_string;
            
                // Find the longest repeating sequence
                std::string longest_sequence = longest_repeating_sequence(input_string);
            
                // Output result
                std::cout << "Longest repeating sequence: " << longest_sequence << std::endl;
            
                return 0;
            }

              bool isValidPassword(const string& password) {

                  bool hasLowercase = false;

                  for (char c : password) {

                          hasUppercase = true;
                      } else if (c >= 'a' && c <= 'z') {
                          hasLowercase = true;
                      } else if (c >= '0' && c <= '9') {
                          hasDigit = true;

                              case '&':
                              case '*':
                              case '(':

                              default:
                                  return false;  Invalid character
                          }
                      }
                  }
              
                   Check if the password has at least one uppercase letter, one lowercase letter, one digit, and one special character
                  return hasUppercase && hasLowercase && hasDigit && hasSpecialChar;

50.Program  that accepts three integers and returns true 
            if one is the middle point between the other two integers, otherwise false

        

        

          
source code

          
            #include < iostream>
              using namespace std; 
              
              bool isMiddlePoint(int a, int b, int c) {
                  // Sort the three integers
                  int maxNum = max(a, max(b, c));
                  int minNum = min(a, min(b, c));
                  int middleNum = a + b + c - maxNum - minNum;
              
                  // Check if middleNum is the middle point between minNum and maxNum
                  return middleNum > minNum && middleNum < maxNum;
              }
              
              int main() {
                 // Variable Declaration // 
                  int a, b, c;
                  cout << "Enter three integers: ";
                  cin >> a >> b >> c;
              
                  if (isMiddlePoint(a, b, c))
                {
                      cout << "One integer is the middle point between the other two.\n";
                  }
                else {
                      cout << "No integer is the middle point between the other two.\n";
                  }
              
                  return 0;
              }

Enter three integers: 44 50 19
            One integer is the middle point between the other two.
