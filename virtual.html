3. Write a C++ program to create an abstract class Shape with abstract methods calculateArea() and calculatePerimeter(). Create derived classes Circle and Triangle that extend the Shape class and implement the respective methods to calculate the area and perimeter of each shape. 

#include <iostream>

        // Abstract base class Shape
        class Shape {
        public:
            // Pure virtual methods make Shape an abstract class
            virtual double calculateArea() const = 0;
            virtual double calculatePerimeter() const = 0;
        };
                
        // Derived class Circle from Shape
        class Circle : public Shape {
        private:
            double radius;
                
        public:
            Circle(double r) : radius(r) {}
                
            double calculateArea() const override {
            return 3.14 * radius * radius; // Area of circle formula
            }
                
            double calculatePerimeter() const override {
                return 2 * 3.14 * radius; // Perimeter of circle formula
            }
        };
                
        // Derived class Triangle from Shape
        class Triangle : public Shape {
        private:
            double side1, side2, side3;
                
        public:
            Triangle(double s1, double s2, double s3) : side1(s1), side2(s2), side3(s3) {}
                
            double calculateArea() const override {
                // Using Heron's formula to calculate the area of a triangle
                double s = (side1 + side2 + side3) / 2; // Semiperimeter
                return sqrt(s * (s - side1) * (s - side2) * (s - side3)); // Area of triangle formula
            }
                
            double calculatePerimeter() const override {
                return side1 + side2 + side3; // Perimeter of triangle formula
            }
        };
                
        int main() {
            // Create a Circle object with radius 5
            Circle circle(5);
            // Create a Triangle object with sides 3, 4, and 5
            Triangle triangle(3, 4, 5);
                
            // Calculate and display area and perimeter of Circle
            std::cout << "Circle Area: " << circle.calculateArea() << std::endl;
            std::cout << "Circle Perimeter: " << circle.calculatePerimeter() << std::endl;
                
            // Calculate and display area and perimeter of Triangle
            std::cout << "Triangle Area: " << triangle.calculateArea() << std::endl;
            std::cout << "Triangle Perimeter: " << triangle.calculatePerimeter() << std::endl;
                
            return 0;
        }

        };

                balance += amount;

#include <iostream>
        #include <cmath>
            
        // Abstract base class Shape3D
        class Shape3D {
        public:
            // Pure virtual methods make Shape3D an abstract class
            virtual double calculateVolume() const = 0;
            virtual double calculateSurfaceArea() const = 0;
        };
            
        // Derived class Sphere from Shape3D
        class Sphere : public Shape3D {
        private:
            double radius;
            
        public:
            // Constructor
            Sphere(double _radius) : radius(_radius) {}
            
            // Implementation of calculateVolume() method for Sphere
            double calculateVolume() const override {
                return (4.0 / 3.0) * 3.14159 * std::pow(radius, 3); // Volume of sphere formula
            }
            
            // Implementation of calculateSurfaceArea() method for Sphere
            double calculateSurfaceArea() const override {
                return 4.0 * 3.14159 * std::pow(radius, 2); // Surface area of sphere formula
            }
        };
            
        // Derived class Cube from Shape3D
        class Cube : public Shape3D {
        private:
            double side;
            
        public:
            // Constructor
            Cube(double _side) : side(_side) {}
            
            // Implementation of calculateVolume() method for Cube
            double calculateVolume() const override {
                return std::pow(side, 3); // Volume of cube formula
            }
            
            // Implementation of calculateSurfaceArea() method for Cube
            double calculateSurfaceArea() const override {
                return 6.0 * std::pow(side, 2); // Surface area of cube formula
            }
        };
            
        int main() {
            // Example usage
            Sphere sphere(5);
            Cube cube(4);
            
            // Calculate and display volume and surface area of Sphere

#include <iostream>
        #include <cmath>
        
        // Abstract base class GeometricShape
        class GeometricShape {
        public:
            // Pure virtual methods make GeometricShape an abstract class
            virtual double area() const = 0;
            virtual double perimeter() const = 0;
        };

        class Triangle : public GeometricShape {
        private:
            double side1, side2, side3;
        
        public:
            // Constructor
            Triangle(double s1, double s2, double s3) : side1(s1), side2(s2), side3(s3) {}
        
            // Implementation of area() method for Triangle using Heron's formula
            double area() const override {
                double s = (side1 + side2 + side3) / 2;
                return sqrt(s * (s - side1) * (s - side2) * (s - side3));
            }
        
            // Implementation of perimeter() method for Triangle
            double perimeter() const override {
                return side1 + side2 + side3;
            }
        };

            // Constructor

            double area() const override {
                return side * side;
            }
        
            // Implementation of perimeter() method for Square
            double perimeter() const override {
                return 4 * side;
            }
        };

            // Example usage
